## OBJECT PROPERTIES
### for in
```
const person = {
  fname:" John",
  lname:" Doe",
  age: 25
};

for (let x in person) {
  txt += person[x];
}
```
### add new properties
```
person.nationality = "English";
```
### delete properties
```
//xóa cả giá trị thuộc tính và thuộc tính đó
const person = {
  firstName: "John",
  lastName: "Doe",
  age: 50,
  eyeColor: "blue"
};

delete person.age;
delete person["age"];
```
